81/*!
82* @brief Task to handle cycling through the pins to take EIT voltage readings
83* @details First waits for initialization of twi communication to complete.
84* Then each round has one channel with an applied current, one grounded, and 14 others.
85* ADCs read all adc channels, discards the two non-read channels, finds the deltaV between appropriate pins
86* and stores all values in the resulting array into a global array after lowering the dataAvailable flag.
87* @param p_params void*, unused.
88*/
89void task_ReadMaterial(void* p_params) {
90 Serial << "Starting Read Material Task" << endl;
91 ADC128D818 ADC_1 (ADC_1ADDRESS);
92 ADC128D818 ADC_2 (ADC_2ADDRESS);
93 CD74HC4067SM Multiplex (s0_PIN,s1_PIN,s2_PIN,s3_PIN,MultiEnable_PIN);
94 PCA9956 CurrCtrl (&Wire);
95
96 const uint8_t maxCurrent = 0xFF;
97
98 uint8_t currPinIndex;
99 double measure[208] = {0}; // Somewhere to store the data for 1 complete measurement
100 double cycleVals[16]; // Somewhere to store the data for 1 energization state
101 double skipCycleVals[14]; // Somewhere to store the important data begining at the correct index for 1 energization state
102 Serial << "Finished initializing Read Material Task" << endl;
103
104 uint8_t cycleIndex = 0;
105 uint8_t state = 0; // Initialization State
106
107 for (;;) {
108 #ifdef DEBUG_READMATERIAL
109 Serial << "Material Task " << state << endl;
110 #endif
111 // Operation requires the TWI devices be initialized, but other tasks include other twi devices.
112 // Priority for this isn't important, so it will just sit in this state until able to initialize.
113 if (state == 0) // Start TWI communication
114 {
115 if (xSemaphoreTake(twiMutex,5) == pdTRUE) // Takes the mutex and returns true if successful
116 {
117 ADC_1.setOperationMode(SINGLE_ENDED); //Mode 1 reads voltages on channels 0-7
118 ADC_1.begin(); // Initializes ADC1
119 ADC_2.setOperationMode(SINGLE_ENDED);//Mode 1 reads voltages on channels 0-7
120 ADC_2.begin(); // Initializes ADC2
121 CurrCtrl.init(PCA9956_ADDRESS,0xFF,false); // Initialize current control address and max brightnes
122 xSemaphoreGive(twiMutex); // Use of twi is complete, free mutex
123 state = 1; // Switch to reading the material
124 }
125 }
126
127 else if (state == 1) // Change Energization State
128 {
129 // cycleIndex initializes to n=0, which is also the grounded electrode.
130 currPinIndex = (cycleIndex + 1) % 16; // The electrode with current applied is always n+1
131
132 // Attempt to send TWI instruction to change LED
133 if (xSemaphoreTake(twiMutex,5) == pdTRUE) // Takes the mutex and returns true if successful
134 {
135 CurrCtrl.offLED(cycleIndex); // Stop producing current for what will become the grounded pin
136 Multiplex.switchPin(cycleIndex); // Use multiplexer to ground the appropriate pin
137 CurrCtrl.onLED(currPinIndex); // Start producing current on the appropriate pin
138 xSemaphoreGive(twiMutex); // Use of twi is complete, free mutex
139 state = 2; // Only advances to start recording if communication was successful
140 }
141 }
142
143 // After a short delay, try to take all the available measurements for 1 energization state
144 else if (state == 2)
145 {
146 if (xSemaphoreTake(twiMutex,5) == pdTRUE) // Takes the mutex and returns true if successful
147 {
148 // Record all ADC channels, electrodes 0-15 are recorded in order by ADC1 7-0 and then ADC2 7-0
149 for (uint8_t i=0;i<8;i++) {
150 cycleVals[i] = ADC_1.readConverted(7-i);
151 cycleVals[8+i] = ADC_2.readConverted(7-i);
152
153 #ifdef DEBUG_READMATERIAL
154 Serial << cycleVals[i] << endl;
155 Serial << cycleVals[8+i] << endl;
156 #endif
157
158 };
159 Serial << "Finished a full round, giving Mutex" << endl;
160 xSemaphoreGive(twiMutex); // Use of twi is complete, free mutex
161 Serial << "Finished a full round and Gave Mutex" << endl;
162
163 // Grounded pin and current pin are not used in analysis so the 0th measurement
164 // should be grounded pin +2 or current pin + 1.
165 for (uint8_t i=0;i<14;i++)
166 {
167 skipCycleVals[i] = cycleVals[(currPinIndex + 1 + i) % 16];
168 #ifdef DEBUG_READMATERIAL
169 Serial << skipCycleVals[i] << endl;
170 #endif
171 };
172 Serial << "end skip" << endl;
173
174 // Finds all the deltaV values used and stores them in a local data storage array,
175 // ordered according to which of the 16 energization states is being recorded.
176 for (uint8_t i=0;i<13;i++)
177 {
178 measure[cycleIndex*13+i] = skipCycleVals[i+1] - skipCycleVals[i];
179 };
180
181 cycleIndex++; // Advances to the next energization state.
182 if (cycleIndex == 16)
183 {
184 Serial << "Finished a full measurement" << endl;
185 cycleIndex = 0; // Resets energization marker to ground pin n=0
186 state = 3; // Change to publish value state
187 }
188 else
189 {
190 state = 1;
191 }
192 }
193 }
194
195 // Pushes all locally stored measurements to a global array accessable by the webpage task
196 else if (state == 3)
197 {
198 #ifdef DEBUG_READMATERIAL
199 Serial << "publishing values" << endl;
200 Serial << dataAvailable.get() << endl;
201 #endif
202 if (dataAvailable.get()) // Checks if the webpage task is currently attempting to read values
203 {
204 dataAvailable.put(false); // Lower the flag to avoid corrupted data
205 #ifdef DEBUG_READMATERIAL
206 Serial << "Took dataMutex" << endl;
207 #endif
208 // Record every datapoint
209 for (uint8_t n=0;n<208;n++)
210 {
211 publish[n] = measure[n];
212 #ifdef DEBUG_READMATERIAL
213 Serial << measure[n] << endl;
214 #endif
215 }
216 dataAvailable.put(true); // data is no longer being used, raise the allow flag
217 if (readVFLG.get() == false) {
218 if (initializeVFLG.get() == false)
219 {
220 initializeVFLG.put(true);
221 }
222 else
223 {
224 readVFLG.put(true);
225 }
226 }
227 #ifdef DEBUG_READMATERIAL
228 Serial << "Gave dataMutex" << endl;
229 #endif
230 state = 1; // go back to energizing the newest state
231 }
232 else {
233 Serial << "Failed to take dataMutex" << endl;
234 }
235 }
236 vTaskDelay(50/portTICK_PERIOD_MS); // Delay 50ms
237 }
238}

240/*!
241* @brief Task to handle controlling the table position using an IMU and two motors
242* @details Has PID control on both motors to attempt to reach the desired setpoint and measured by a BNO055A IMU.
243* @param p_params void*, unused.
244*/
245void task_controlMotors(void *parameter) {
246 float x_angle, y_angle, xTargetAngle, yTargetAngle, xAngleErr, yAngleErr, errSumX,errSumY, effX, effY;
247 float xAngleErrOld,yAngleErrOld;
248 int16_t encoderXTicks, encoderYTicks, errX, errY;
249 uint8_t pwm_X, pwm_Y;
250 const float KP = 40; // Proportional gain for speed control
251 const float KI = 0.1; // Integral gain for speed control
252 const float KD = 10; // Integral gain for speed control
253
254 uint8_t maxAngle = 10; // software limit for desired angle
255 xTargetAngle = 0.0; // Target angle is 0 degrees by default
256 yTargetAngle = 0.0; // Target angle is 0 degrees by default
257
258 MOTOR_brake(motorXPin1, motorXPin2, 0, 1); // Initially stop both motors
259 MOTOR_brake(motorYPin1, motorYPin2, 2, 3);
260
261 uint8_t state = 0;
262 for (;;) {
263 // Initial State waits for IMU to initialize
264 if (state == 0) {
265 if (xSemaphoreTake(twiMutex,5) == pdTRUE) // Takes the mutex and returns true if successful
266 {
267 // Only advance states if IMU sucessfully initializes
268 if (!IMU_init(5)) {
269 Serial.println("Failed to initialize IMU!");
270 }
271 else {
272 Serial.println("IMU initialized.");
273 state = 1;
274 }
275 xSemaphoreGive(twiMutex);
276 }
277 }
278 // After Initialization, control the motor according to setpoint
279 else if (state == 1) {
280 /* get the currentl angles of the platform*/
281 if (xSemaphoreTake(twiMutex,5) == pdTRUE)
282 {
283 IMU_getAngles(x_angle, y_angle);
284 xSemaphoreGive(twiMutex);
285 }
286 // if (abs(x_angle) > 15.0f || abs(y_angle) > 15.0f) {
287 // // If tilt angle exceeds 15 degrees, stop motors for safety
288 // MOTOR_brake(motorXPin1, motorXPin2, 0, 1);
289 // MOTOR_brake(motorYPin1, motorYPin2, 2, 3);
290 // Serial.println("Tilt angle too high! Motors stopped for safety.");
291 // vTaskDelay(500 / portTICK_PERIOD_MS); // Delay for 500 ms
292 // continue; // Skip the rest of the loop
293 // }
294
295 // Adopts targets communicated by the webpage task
296 xTargetAngle = xBar.get()*maxAngle; // Centroid values communicated are from -1 to 1
297 yTargetAngle = yBar.get()*maxAngle;
298
299 // Determine error
300 xAngleErr = xTargetAngle - x_angle;
301 yAngleErr = yTargetAngle - y_angle;
302
303 // Constrain Integral Error to produce at most max effort to eliminate run-away integral control
304 errSumX = constrain(errSumX + xAngleErr,-2550,2550);
305 errSumY = constrain(errSumY + yAngleErr,-2550,2550);
306 #ifdef DEBUG
307 Serial << endl;
308 Serial << xAngleErr << " " << yAngleErr << endl;
309 Serial << errSumX << " " << errSumY << endl;
310 #endif
311 // Calculate appropriate effort
312 effX = xAngleErr*KP + errSumX*KI + (xAngleErr-xAngleErrOld)*KD;
313 effY = yAngleErr*KP + errSumY*KI + (yAngleErr-yAngleErrOld)*KD;
314
315 // Assign previous error for use in derivative control
316 xAngleErrOld = xAngleErr;
317 yAngleErrOld = yAngleErr;
318
319 // constrain effort to maximum allowed
320 pwm_X = constrain((abs(effX)), 0, 255);
321 pwm_Y = constrain((abs(effY)), 0, 255);
322
323 #ifdef DEBUG
324 Serial.print(pwm_X);
325 Serial.print(" ");
326 Serial.println(pwm_Y);
327 #endif
328
329 /* control motor X*/
330 if (effX > 0) {
331 MOTOR_forward(motorXPin1, motorXPin2, 0, 1, pwm_X);
332 } else if (effX < 0) {
333 MOTOR_reverse(motorXPin1, motorXPin2, 0, 1, pwm_X);
334 } else {
335 MOTOR_brake(motorXPin1, motorXPin2, 0, 1); // Shouldn't get here
336 }
337 /* control motor Y*/
338 if (effY > 0) {
339 MOTOR_reverse(motorYPin1, motorYPin2, 2, 3, pwm_Y);
340 } else if (effY < 0) {
341 MOTOR_forward(motorYPin1, motorYPin2, 2, 3, pwm_Y);
342 } else {
343 MOTOR_brake(motorYPin1, motorYPin2, 2, 3); // Shouldn't get here
344 }
345 }
346 vTaskDelay(5/portTICK_PERIOD_MS); // Delay for 5 ms
347 }
