11/**
12 * @class CD74HC4067SM
13 * @brief A class to control a CD74HC4067SM 1:16 analog multiplexer/demultiplexer chip.
14 *
15 * This class provides a convenient interface for controlling a CD74HC4067SM multiplexer,
16 * which allows selecting one of 16 analog channels using 4 binary select pins (S0-S3)
17 * and an active-low enable pin. The chip is commonly used in sensor arrays or to expand
18 * the number of analog inputs available on a microcontroller.
19 *
20* @param Spin0 the Arduino pin used to control the s0 pin
21* @param Spin1 the Arduino pin used to control the s1 pin
22* @param Spin2 the Arduino pin used to control the s2 pin
23* @param Spin3 the Arduino pin used to control the s3 pin
24* @param enablePin the Arduino pin used to control !E pin
25 *
26 * @details The multiplexer uses binary addressing via pins S0, S1, S2, and S3 to select
27 * one of 16 channels (0-15). When enabled (nE = LOW), the selected channel is connected
28 * to the common I/O pin. When disabled (nE = HIGH), no channels are connected.
29 *
30 * @see switchPin(), enable(), disable()
31 */
32CD74HC4067SM::CD74HC4067SM(uint8_t SPin0,
33 uint8_t SPin1,
34 uint8_t SPin2,
35 uint8_t SPin3,
36 uint8_t enablePin)
37{
38 // Assign all pins in use
39 s0 = SPin0;
40 s1 = SPin1;
41 s2 = SPin2;
42 s3 = SPin3;

53/*! @brief switches connection to chosen pin value
54* @details enables the set of s-pins to represent the binary
55* of the chosen pin which closes the selected circuit to the chip output
56* @param pin the 0-15 value of the desired channel
57*/
58void CD74HC4067SM::switchPin(uint8_t pin)
59{
60 // s0 Controls bit 0, check if the requested pin
61 // needs s0 to be HIGH, otherwise set it LOW
62 if (pin & 0001) {digitalWrite(s0,HIGH);}
63 else {digitalWrite(s0,LOW);}
64
65 // s1 Controls bit 1, check if the requested pin
66 // needs s1 to be HIGH, otherwise set it LOW
67 if (pin & 0010) {digitalWrite(s1,HIGH);}
68 else { digitalWrite(s1,LOW);}
69
70 // s2 Controls bit 2, check if the requested pin
71 // needs s2 to be HIGH, otherwise set it LOW
72 if (pin & 0100) {digitalWrite(s2,HIGH);}

75 // s3 Controls bit 3, check if the requested pin
