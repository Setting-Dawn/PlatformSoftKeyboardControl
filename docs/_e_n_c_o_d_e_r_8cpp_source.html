9/**
10 * @brief Initialize an encoder with full quadrature decoding.
11 *
12 * Configures the ESP32 encoder to use internal weak pull-up resistors and
13 * attaches it to the specified pins for full quadrature counting. If inverted
14 * flag is set, the encoder count direction is reversed.
15 *
16 * @param encoder Reference to the ESP32Encoder object to initialize
17 * @param pinA First GPIO pin for encoder input (typically Phase A)
18 * @param pinB Second GPIO pin for encoder input (typically Phase B)
19 * @param inverted Boolean flag to reverse the count direction:
20 * true = negate the current count for inverted direction
21 * false = use normal counting direction
22 *
23 * @return void
24 *
25 * @details Enables ESP32Encoder internal weak pull-up resistors and uses
26 * full quadrature decoding for 4x count resolution.
27 *
28 * @see ENCODER_getCount(), ENCODER_zero()
29 */
30void ENCODER_init(ESP32Encoder &encoder, uint8_t pinA, uint8_t pinB, bool inverted) {
31 ESP32Encoder::useInternalWeakPullResistors = puType ::up;
32 encoder.attachFullQuad(pinA, pinB);
33 if (inverted) {
34 encoder.setCount(-encoder.getCount());
35 }
36}

38/**
39 * @brief Retrieve the current encoder count value.
40 *
41 * Returns the current count from the ESP32 encoder. The count represents
42 * the accumulated encoder position and may be positive or negative depending
43 * on the direction of rotation and whether the encoder was inverted.
44 *
45 * @param encoder Reference to the ESP32Encoder object to read from
46 *
47 * @return int64_t The current encoder count value
48 *
49 * @details This function reads the counter without modifying its state.
50 * For full quadrature decoding, each mechanical detent may count as multiple
51 * counts depending on the encoder type (typically 4 counts per detent).
52 *
53 * @see ENCODER_init(), ENCODER_zero()
54 */
55int64_t ENCODER_getCount(ESP32Encoder &encoder) {
56 return encoder.getCount();
57}
